# AWS Terraform Configuration for AgentMap Serverless Functions
# Corrected version addressing critical deployment issues

terraform {
  required_version = ">= 1.0"
  required_providers {
    aws = {
      source  = "hashicorp/aws"
      version = "~> 5.0"
    }
  }
}

provider "aws" {
  region = var.aws_region
}

# Variables
variable "aws_region" {
  description = "AWS region"
  type        = string
  default     = "us-east-1"
}

variable "project_name" {
  description = "Project name for resource naming"
  type        = string
  default     = "agentmap"
}

variable "environment" {
  description = "Environment (dev, staging, prod)"
  type        = string
  default     = "dev"
}

variable "lambda_timeout" {
  description = "Lambda function timeout in seconds"
  type        = number
  default     = 300
}

variable "lambda_memory_size" {
  description = "Lambda function memory size in MB"
  type        = number
  default     = 512
}

variable "create_api_gateway" {
  description = "Whether to create API Gateway (set to false for event-only deployment)"
  type        = bool
  default     = true
}

variable "create_full_stack" {
  description = "Whether to create full stack with SQS, DynamoDB, etc. (false for minimal deployment)"
  type        = bool
  default     = false
}

# Local variables
locals {
  function_name = "${var.project_name}-${var.environment}"
  common_tags = {
    Project     = var.project_name
    Environment = var.environment
    ManagedBy   = "terraform"
  }
}

# Lambda execution role
resource "aws_iam_role" "lambda_execution_role" {
  name = "${local.function_name}-lambda-role"

  assume_role_policy = jsonencode({
    Version = "2012-10-17"
    Statement = [
      {
        Action = "sts:AssumeRole"
        Effect = "Allow"
        Principal = {
          Service = "lambda.amazonaws.com"
        }
      }
    ]
  })

  tags = local.common_tags
}

# Basic Lambda execution policy
resource "aws_iam_role_policy_attachment" "lambda_basic" {
  policy_arn = "arn:aws:iam::aws:policy/service-role/AWSLambdaBasicExecutionRole"
  role       = aws_iam_role.lambda_execution_role.name
}

# Additional permissions for full stack
resource "aws_iam_role_policy" "lambda_additional_policy" {
  count = var.create_full_stack ? 1 : 0
  name  = "${local.function_name}-additional-policy"
  role  = aws_iam_role.lambda_execution_role.id

  policy = jsonencode({
    Version = "2012-10-17"
    Statement = [
      {
        Effect = "Allow"
        Action = [
          "sqs:ReceiveMessage",
          "sqs:DeleteMessage",
          "sqs:GetQueueAttributes"
        ]
        Resource = aws_sqs_queue.agentmap_queue[0].arn
      },
      {
        Effect = "Allow"
        Action = [
          "dynamodb:DescribeStream",
          "dynamodb:GetRecords",
          "dynamodb:GetShardIterator",
          "dynamodb:ListStreams"
        ]
        Resource = aws_dynamodb_table.agentmap_table[0].stream_arn
      },
      {
        Effect = "Allow"
        Action = [
          "s3:GetObject",
          "s3:PutObject"
        ]
        Resource = "${aws_s3_bucket.agentmap_data[0].arn}/*"
      },
      {
        Effect = "Allow"
        Action = [
          "sns:Publish"
        ]
        Resource = aws_sns_topic.agentmap_results[0].arn
      }
    ]
  })
}

# Lambda function package - FIXED: Package only deployment directory
data "archive_file" "lambda_zip" {
  type        = "zip"
  source_dir  = "${path.module}"
  output_path = "${path.module}/lambda_package.zip"
  excludes = [
    "*.tf",
    "*.txt",
    "README.md",
    "lambda_package.zip",
    ".terraform*",
    "terraform.tfstate*",
    "aws-terraform-deployment.txt"
  ]
}

# Main Lambda function - FIXED: Correct handler path and Python version
resource "aws_lambda_function" "agentmap_function" {
  filename         = data.archive_file.lambda_zip.output_path
  function_name    = local.function_name
  role            = aws_iam_role.lambda_execution_role.arn
  handler         = "main.handler"  # FIXED: Correct handler path
  source_code_hash = data.archive_file.lambda_zip.output_base64sha256
  runtime         = "python3.12"   # FIXED: Updated Python version
  timeout         = var.lambda_timeout
  memory_size     = var.lambda_memory_size

  environment {
    variables = {
      AGENTMAP_ENVIRONMENT = var.environment
      AGENTMAP_LOG_LEVEL  = "INFO"
      AGENTMAP_DATA_BUCKET = var.create_full_stack ? aws_s3_bucket.agentmap_data[0].id : ""
      AGENTMAP_RESULTS_TOPIC = var.create_full_stack ? aws_sns_topic.agentmap_results[0].arn : ""
    }
  }

  tags = local.common_tags

  depends_on = [
    aws_iam_role_policy_attachment.lambda_basic,
    aws_cloudwatch_log_group.lambda_logs,
  ]
}

# CloudWatch Log Group
resource "aws_cloudwatch_log_group" "lambda_logs" {
  name              = "/aws/lambda/${local.function_name}"
  retention_in_days = 14
  tags              = local.common_tags
}

# API Gateway for HTTP triggers (optional)
resource "aws_apigatewayv2_api" "agentmap_api" {
  count         = var.create_api_gateway ? 1 : 0
  name          = "${local.function_name}-api"
  protocol_type = "HTTP"
  description   = "AgentMap HTTP API"
  
  cors_configuration {
    allow_credentials = false
    allow_headers     = ["content-type", "x-amz-date", "authorization"]
    allow_methods     = ["*"]
    allow_origins     = ["*"]
    max_age          = 86400
  }

  tags = local.common_tags
}

resource "aws_apigatewayv2_stage" "agentmap_api_stage" {
  count       = var.create_api_gateway ? 1 : 0
  api_id      = aws_apigatewayv2_api.agentmap_api[0].id
  name        = var.environment
  auto_deploy = true
  tags        = local.common_tags
}

resource "aws_apigatewayv2_integration" "agentmap_api_integration" {
  count                = var.create_api_gateway ? 1 : 0
  api_id               = aws_apigatewayv2_api.agentmap_api[0].id
  integration_type     = "AWS_PROXY"
  connection_type      = "INTERNET"
  integration_method   = "POST"
  integration_uri      = aws_lambda_function.agentmap_function.invoke_arn
  passthrough_behavior = "WHEN_NO_MATCH"
}

resource "aws_apigatewayv2_route" "agentmap_api_route" {
  count     = var.create_api_gateway ? 1 : 0
  api_id    = aws_apigatewayv2_api.agentmap_api[0].id
  route_key = "$default"
  target    = "integrations/${aws_apigatewayv2_integration.agentmap_api_integration[0].id}"
}

resource "aws_lambda_permission" "api_gateway_invoke" {
  count         = var.create_api_gateway ? 1 : 0
  statement_id  = "AllowExecutionFromAPIGateway"
  action        = "lambda:InvokeFunction"
  function_name = aws_lambda_function.agentmap_function.function_name
  principal     = "apigateway.amazonaws.com"
  source_arn    = "${aws_apigatewayv2_api.agentmap_api[0].execution_arn}/*/*"
}

# SQS Queue for message triggers
resource "aws_sqs_queue" "agentmap_queue" {
  count                     = var.create_full_stack ? 1 : 0
  name                      = "${local.function_name}-queue"
  delay_seconds             = 0
  max_message_size          = 262144
  message_retention_seconds = 1209600  # 14 days
  receive_wait_time_seconds = 20       # Long polling

  tags = local.common_tags
}

resource "aws_sqs_queue" "agentmap_dlq" {
  count = var.create_full_stack ? 1 : 0
  name  = "${local.function_name}-dlq"
  tags  = local.common_tags
}

resource "aws_sqs_queue_redrive_policy" "agentmap_queue_redrive" {
  count     = var.create_full_stack ? 1 : 0
  queue_url = aws_sqs_queue.agentmap_queue[0].id
  redrive_policy = jsonencode({
    deadLetterTargetArn = aws_sqs_queue.agentmap_dlq[0].arn
    maxReceiveCount     = 3
  })
}

# Lambda trigger for SQS
resource "aws_lambda_event_source_mapping" "sqs_trigger" {
  count            = var.create_full_stack ? 1 : 0
  event_source_arn = aws_sqs_queue.agentmap_queue[0].arn
  function_name    = aws_lambda_function.agentmap_function.arn
  batch_size       = 1
  
  maximum_batching_window_in_seconds = 5
  
  depends_on = [aws_iam_role_policy.lambda_additional_policy]
}

# DynamoDB table for database triggers
resource "aws_dynamodb_table" "agentmap_table" {
  count            = var.create_full_stack ? 1 : 0
  name             = "${local.function_name}-table"
  billing_mode     = "PAY_PER_REQUEST"
  hash_key         = "id"
  stream_enabled   = true
  stream_view_type = "NEW_AND_OLD_IMAGES"

  attribute {
    name = "id"
    type = "S"
  }

  tags = local.common_tags
}

# Lambda trigger for DynamoDB Streams
resource "aws_lambda_event_source_mapping" "dynamodb_trigger" {
  count             = var.create_full_stack ? 1 : 0
  event_source_arn  = aws_dynamodb_table.agentmap_table[0].stream_arn
  function_name     = aws_lambda_function.agentmap_function.arn
  starting_position = "LATEST"
  batch_size        = 1

  depends_on = [aws_iam_role_policy.lambda_additional_policy]
}

# S3 bucket for storage triggers
resource "aws_s3_bucket" "agentmap_data" {
  count  = var.create_full_stack ? 1 : 0
  bucket = "${local.function_name}-data-${random_id.bucket_suffix[0].hex}"
  tags   = local.common_tags
}

resource "random_id" "bucket_suffix" {
  count       = var.create_full_stack ? 1 : 0
  byte_length = 4
}

resource "aws_s3_bucket_versioning" "agentmap_data_versioning" {
  count  = var.create_full_stack ? 1 : 0
  bucket = aws_s3_bucket.agentmap_data[0].id
  versioning_configuration {
    status = "Enabled"
  }
}

resource "aws_s3_bucket_notification" "agentmap_data_notification" {
  count  = var.create_full_stack ? 1 : 0
  bucket = aws_s3_bucket.agentmap_data[0].id

  lambda_function {
    lambda_function_arn = aws_lambda_function.agentmap_function.arn
    events              = ["s3:ObjectCreated:*"]
    filter_prefix       = "uploads/"
    filter_suffix       = ".csv"
  }

  depends_on = [aws_lambda_permission.s3_invoke]
}

resource "aws_lambda_permission" "s3_invoke" {
  count         = var.create_full_stack ? 1 : 0
  statement_id  = "AllowExecutionFromS3Bucket"
  action        = "lambda:InvokeFunction"
  function_name = aws_lambda_function.agentmap_function.function_name
  principal     = "s3.amazonaws.com"
  source_arn    = aws_s3_bucket.agentmap_data[0].arn
}

# SNS topic for results publishing
resource "aws_sns_topic" "agentmap_results" {
  count = var.create_full_stack ? 1 : 0
  name  = "${local.function_name}-results"
  tags  = local.common_tags
}

# EventBridge rule for scheduled triggers
resource "aws_cloudwatch_event_rule" "agentmap_schedule" {
  count               = var.create_full_stack ? 1 : 0
  name                = "${local.function_name}-schedule"
  description         = "Trigger AgentMap function on schedule"
  schedule_expression = "rate(1 hour)"
  tags                = local.common_tags
}

resource "aws_cloudwatch_event_target" "lambda_target" {
  count     = var.create_full_stack ? 1 : 0
  rule      = aws_cloudwatch_event_rule.agentmap_schedule[0].name
  target_id = "AgentMapLambdaTarget"
  arn       = aws_lambda_function.agentmap_function.arn

  input = jsonencode({
    action = "run"
    graph  = "ScheduledGraph"
    scheduled_event = {
      detail_type = "Scheduled Event"
      source      = "aws.events"
    }
  })
}

resource "aws_lambda_permission" "eventbridge_invoke" {
  count         = var.create_full_stack ? 1 : 0
  statement_id  = "AllowExecutionFromEventBridge"
  action        = "lambda:InvokeFunction"
  function_name = aws_lambda_function.agentmap_function.function_name
  principal     = "events.amazonaws.com"
  source_arn    = aws_cloudwatch_event_rule.agentmap_schedule[0].arn
}

# CloudWatch Dashboard
resource "aws_cloudwatch_dashboard" "agentmap_dashboard" {
  count          = var.create_full_stack ? 1 : 0
  dashboard_name = "${local.function_name}-dashboard"

  dashboard_body = jsonencode({
    widgets = [
      {
        type   = "metric"
        x      = 0
        y      = 0
        width  = 12
        height = 6

        properties = {
          metrics = [
            ["AWS/Lambda", "Duration", "FunctionName", aws_lambda_function.agentmap_function.function_name],
            [".", "Errors", ".", "."],
            [".", "Invocations", ".", "."],
            [".", "Throttles", ".", "."]
          ]
          view    = "timeSeries"
          stacked = false
          region  = var.aws_region
          title   = "Lambda Metrics"
          period  = 300
        }
      },
      {
        type   = "log"
        x      = 0
        y      = 6
        width  = 24
        height = 6

        properties = {
          query = "SOURCE '/aws/lambda/${aws_lambda_function.agentmap_function.function_name}'\n| fields @timestamp, @message\n| filter @message like /ERROR/\n| sort @timestamp desc\n| limit 20"
          region = var.aws_region
          title  = "Recent Errors"
        }
      }
    ]
  })
}
